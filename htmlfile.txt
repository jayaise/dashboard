  <h1>MQTT WebSocket Test</h1>
  <button id="send">Send WS MQTT Message</button>

  <script>
    const client = mqtt.connect("ws://localhost:15675/ws", {
      username: "admin",
      password: "admin123",
    });

    client.on("connect", function () {
      console.log("Browser WS MQTT connected");
    });

    document.getElementById("send").onclick = function () {
      client.publish("devices/gw1/health", "hello-from-browser-ws");
      console.log("Sent!");
    };
  </script>
